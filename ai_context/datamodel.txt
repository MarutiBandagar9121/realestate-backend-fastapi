// ───────────────────────────────────────
// RELATIONS
// ───────────────────────────────────────

// Property core
property_type.id < property.property_type_id
property_type.id < property_type_config.property_type_id

// Node tree
property.id < property_node.property_id
property_node_type.id < property_node.node_type_id
property_node.id < property_node.parent_node_id

// Node details (each node has exactly one detail record)
property_node.id - building_details.node_id
property_node.id - wing_details.node_id
property_node.id - floor_details.node_id
property_node.id - unit_details.node_id
property_node.id - land_details.node_id
property_node.id - industrial_details.node_id
property_node.id - logistics_details.node_id

// Offerings
property_node.id < offering.node_id
property.id < offering.property_id
offering.id - offering_lease_detail.offering_id
offering.id - offering_sale_detail.offering_id
offering.id - offering_fractional_detail.offering_id

// Coworking
property_node.id < coworking_product.node_id
property.id < coworking_product.property_id

// Parties
party.id < property.operator_id
party.id < unit_details.owner_id
